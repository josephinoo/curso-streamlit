app.py — El Director de Orquesta

import streamlit as st
from auth import show_login

# 1. Configuracion global — SIEMPRE primero
st.set_page_config(
    page_title="RetailDash",
    page_icon="📈",
    layout="wide",
)

# 2. Inicializar session_state (solo la primera vez)
if "logged_in" not in st.session_state:
    st.session_state.logged_in = False
    st.session_state.nombre    = ""
    st.session_state.rol       = ""

# 3. Definir las paginas del dashboard
dashboard_pages = [
    st.Page("pages/1_Resumen.py",
            title="Resumen", icon="📊", default=True),
    st.Page("pages/2_Ventas.py",
            title="Ventas", icon="💰"),
    st.Page("pages/3_Geografico.py",
            title="Geografico", icon="🗺️"),
    st.Page("pages/4_Datos.py",
            title="Datos", icon="📁"),
]

# 4. Routing segun el estado
if not st.session_state.logged_in:
    pg = st.navigation(
        [st.Page(show_login, title="Login")],
        position="hidden",   # sin sidebar en el login
    )
else:
    pg = st.navigation(dashboard_pages)

pg.run()
Flujo de ejecución
1
Python ejecuta app.py de arriba a abajo
↓
2
logged_in es False → muestra login
↓ usuario hace login
3
logged_in = True en session_state
↓ st.rerun()
4
app.py corre otra vez → muestra dashboard
📌 set_page_config SOLO va en app.py. Si lo pones en otro archivo, da error.

Clave: que entiendan que app.py se re-ejecuta completo en cada interacción. session_state es lo único que “sobrevive” entre ejecuciones.

¿Por qué necesitamos session_state?

El problema — sin session_state
# Streamlit re-ejecuta el script COMPLETO
# con cada interacción del usuario.
# Una variable normal no sobrevive:

logged_in = True       # ← se guarda

# ... usuario hace click en algo ...
# ... Python vuelve a ejecutar todo ...

logged_in = False      # ← volvió al inicio :(
# El usuario tiene que volver a loguearse!
🔄 Streamlit no es como un programa normal — re-ejecuta todo el script cada vez que el usuario interactúa.
La solución — session_state
# session_state es como una pizarra que
# NO se borra entre re-ejecuciones.
# Vive mientras el usuario tiene la app abierta.

# Escribir:
st.session_state.logged_in = True
st.session_state.nombre    = "Ana García"

# Leer:
if st.session_state.logged_in:
    st.write(f"Hola {st.session_state.nombre}")

# Verificar si existe (inicialización):
if "logged_in" not in st.session_state:
    st.session_state.logged_in = False
✅ Analogía: es una nota adhesiva pegada al navegador del usuario. Persiste aunque el script se vuelva a ejecutar.

auth.py — La Pantalla de Login

import streamlit as st

USUARIOS = {
    "admin":  {"password": "admin123",
               "nombre":   "Administrador",
               "rol":      "Admin"},
    "ana":    {"password": "ana2024",
               "nombre":   "Ana Garcia",
               "rol":      "Analista"},
}

def verificar_credenciales(usuario, password):
    if usuario in USUARIOS:
        return USUARIOS[usuario]["password"] == password
    return False

def show_login():
    # Centrar con columnas — truco clasico
    _, col, _ = st.columns([1, 1.5, 1])

    with col:
        st.title("🛒 RetailDash")
        st.subheader("Iniciar sesion")
        st.divider()

        with st.form("login_form"):
            usuario  = st.text_input("Usuario")
            password = st.text_input(
                "Contrasena", type="password"
            )
            submit = st.form_submit_button(
                "Iniciar sesion",
                use_container_width=True,
                type="primary",
            )

        if submit:
            u = usuario.strip().lower()
            if verificar_credenciales(u, password):
                st.session_state.logged_in = True
                st.session_state.nombre = USUARIOS[u]["nombre"]
                st.session_state.rol    = USUARIOS[u]["rol"]
                st.rerun()  # va al dashboard
            else:
                st.error("Usuario o contrasena incorrectos.")

❌ Sin st.form — problema

# dispara una re-ejecucion completa.
# En login esto no es grave, pero en

    password = st.text_input("Password",
                              type="password")

    # Este boton SI dispara el rerun

        "Iniciar sesion",
        type="primary"
    )

if submit:
    # aqui procesamos
    verificar(usuario, password)

data_loader.py — Carga y Limpieza

import pandas as pd
import numpy as np
import streamlit as st

REGION_COORDS = {
    "North":   {"lat": 43.0, "lon": -85.0},
    "South":   {"lat": 30.0, "lon": -90.0},
    "East":    {"lat": 40.7, "lon": -74.0},
    "West":    {"lat": 37.8, "lon": -120.5},
    "Central": {"lat": 41.8, "lon": -93.0},
}

@st.cache_data(show_spinner="Cargando datos...")
def cargar_datos(ruta="data/retail_sales.csv"):
    try:
        df = pd.read_csv(ruta, parse_dates=["Date"])
    except FileNotFoundError:
        st.warning("CSV no encontrado. Usando demo.")
        df = _generar_datos_demo()

    # Limpieza 1: eliminar "NaN?" en Category
    df = df[~df["Category"].astype(str).str.strip()
              .isin(["NaN?", "nan", "NaN", ""])]
    df = df.dropna(subset=["Category"])

    # Limpieza 2: nulos en columnas criticas
    df = df.dropna(
        subset=["Sales","Profit","Quantity","Date","Region"]
    )

    # Limpieza 3: negativos (anomalias)
    df = df[(df["Sales"] >= 0) & (df["Quantity"] >= 0)]

    # Columnas derivadas
    df["Month"] = df["Date"].dt.to_period("M").astype(str)
    df["Year"]  = df["Date"].dt.year
    df["Profit_Margin"] = (
        df["Profit"] / df["Sales"].replace(0, np.nan)
    ) * 100

    return df.reset_index(drop=True)
Los 3 pasos de limpieza
1
Eliminar "NaN?" — la cadena de texto literal que el dataset tiene en Category. No es un nulo real, hay que buscarlo manualmente
2
Eliminar nulos en columnas críticas. Si falta Sales o Date, esa fila no sirve
3
Eliminar negativos en Sales y Quantity — son errores de carga de datos
🎯 @st.cache_data hace que la limpieza ocurra una sola vez. Las siguientes llamadas devuelven el resultado guardado en memoria — instantáneo.

utils.py — La Librería Interna

def sidebar_con_filtros():
    """Funcion que llaman todas las paginas al inicio.
    1. Verifica que el usuario este logueado
    2. Muestra sidebar con logout y filtros
    3. Devuelve el DataFrame filtrado
    """
    # Guard: redirigir si no hay sesion
    if not st.session_state.get("logged_in", False):
        st.warning("Debes iniciar sesion.")
        st.stop()

    with st.sidebar:
        st.title("🛒 RetailDash")
        st.write(f"**{st.session_state.nombre}**")
        st.caption(f"Rol: {st.session_state.rol}")
        st.divider()

        if st.button("Cerrar sesion",
                     use_container_width=True):
            st.session_state.logged_in = False
            st.rerun()

        st.divider()
        st.subheader("Filtros")
        df_raw = cargar_datos()

        categorias = ["Todas"] + sorted(
            df_raw["Category"].unique().tolist()
        )
        cat_sel = st.selectbox(
            "Categoria", categorias, key="cat_sel"
        )
        regiones = ["Todas"] + sorted(
            df_raw["Region"].unique().tolist()
        )
        reg_sel = st.selectbox(
            "Region", regiones, key="reg_sel"
        )
        anos = sorted(df_raw["Year"].unique().tolist())
        ano_sel = st.multiselect(
            "Ano", anos, default=anos, key="ano_sel"
        )

    df = df_raw.copy()
    if cat_sel != "Todas":
        df = df[df["Category"] == cat_sel]
    if reg_sel != "Todas":
        df = df[df["Region"] == reg_sel]
    if ano_sel:
        df = df[df["Year"].isin(ano_sel)]

    if df.empty:
        st.warning("Sin datos para estos filtros.")
        st.stop()

    return df

Los KPIs — st.metric

def calcular_kpis(df):
    total_ventas  = df["Sales"].sum()
    total_profit  = df["Profit"].sum()
    total_items   = int(df["Quantity"].sum())
    num_registros = len(df)
    margen_prom   = (total_profit / total_ventas
                     * 100) if total_ventas else 0
    ticket_prom   = (total_ventas / num_registros
                     ) if num_registros else 0
    return {
        "ventas": total_ventas,
        "profit": total_profit,
        "items":  total_items,
        "registros": num_registros,
        "margen":  margen_prom,
        "ticket":  ticket_prom,
    }

def mostrar_kpis(df):
    kpis = calcular_kpis(df)
    k1, k2, k3, k4, k5, k6 = st.columns(6)

    k1.metric("Ventas Totales",

import plotly.graph_objects as go
import plotly.express as px

# Patron: recibir df, devolver Figure
def grafico_ventas_tiempo(df):
    # 1. Agrupar los datos
    ts = df.groupby("Month")["Sales"].sum().reset_index()
    ts["Month_dt"] = pd.to_datetime(ts["Month"])
    ts = ts.sort_values("Month_dt")

    # 2. Crear la figura
    fig = go.Figure()
    fig.add_trace(go.Scatter(
        x=ts["Month_dt"], y=ts["Sales"],
        mode="lines+markers",
        line=dict(color="#FF4B4B", width=2.5),
        fill="tozeroy",
        fillcolor="rgba(255,75,75,0.10)",
        name="Ventas",
    ))

    # 3. Ajustar titulo y ejes
    fig.update_layout(
        title="Ventas Mensuales",
        xaxis_title="",
        yaxis_title="USD"
    )
    return fig

# En la pagina, mostrar con:
st.plotly_chart(
    grafico_ventas_tiempo(df),
    use_container_width=True
)

El Mapa — go.Scattergeo

def grafico_mapa_regiones(df):
    # Agregar ventas y ganancia por region
    reg = df.groupby("Region").agg(
        Sales=("Sales",    "sum"),
        Profit=("Profit",  "sum"),
        Registros=("Sales","count"),
    ).reset_index()

    # Asignar coordenadas del diccionario
    reg["lat"] = reg["Region"].map(
        lambda r: REGION_COORDS.get(r,{}).get("lat",0)
    )
    reg["lon"] = reg["Region"].map(
        lambda r: REGION_COORDS.get(r,{}).get("lon",0)
    )

    fig = go.Figure(go.Scattergeo(
        lat=reg["lat"],
        lon=reg["lon"],
        text=reg.apply(
            lambda r: (
                f"{r['Region']}
"
                f"Ventas: {fmt_money(r['Sales'])}
"
                f"Ganancia: {fmt_money(r['Profit'])}"
            ), axis=1
        ),
        hoverinfo="text",
        mode="markers",
        marker=dict(
            # burbuja proporcional a ventas
            size=(reg["Sales"]/reg["Sales"].max()*45+15),
            color=reg["Sales"],
            colorscale="Reds",
            showscale=True,
            line=dict(color="white", width=1),
        ),
    ))

    fig.update_geos(scope="usa")
    fig.update_layout(
        title="Mapa de Ventas por Region (EE.UU.)",
        height=480
    )
    return fig
Conceptos clave del mapa
1
Coordenadas manuales — el dataset no tiene GPS. Las definimos en REGION_COORDS en data_loader.py
2
Tamaño proporcional — Sales/max*45+15 escala el radio entre 15px (mínimo) y 60px (máximo)
3
scope="usa" — limita el mapa a EE.UU. Podría ser "world", "europe", etc.
4
Hover text con HTML — <b> y <br> funcionan dentro del texto del tooltip
🗺️ Scattergeo es la forma más simple de hacer mapas en Plotly. No necesita API keys ni tiles externos.

Errores Frecuentes y Soluciones

!
ModuleNotFoundError: plotly
pip install plotly
!
set_page_config() can only be called once
Solo en app.py, no en los archivos de pages/
!
StreamlitDuplicateWidgetId
Agregar key= a todos los widgets de la sidebar
!
FileNotFoundError: retail_sales.csv
La app usa datos demo automaticamente. No es un error bloqueante
!
La app recarga sin parar (loop infinito)
st.rerun() solo dentro de un if condicional, nunca en el flujo principal
Checklist antes de ejecutar
✓ Checklist
Carpeta pages/ existe con los 4 archivos
Carpeta data/ existe (con o sin el CSV)
set_page_config solo en app.py
Todos los widgets de sidebar tienen key=
Cada página llama sidebar_con_filtros() primero
Dependencias instaladas con pip install -r requirements.txt
